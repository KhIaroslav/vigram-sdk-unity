    Installation And First Run

    Install Through Package Manager

    In Unity, open Window > Package Manager, press +, select Add package from git URL..., and enter:

    https://github.com/vigram-sw/sdk-unity-vidoc-distribution.git?path=/Packages/com.vigram.sdk-gh1#1.0.0

    You can also add the dependency directly to Packages/manifest.json:

    {
  "dependencies": {
    "com.vigram.sdk-gh1": "https://github.com/vigram-sw/sdk-unity-vidoc-distribution.git?path=/Packages/com.vigram.sdk-gh1#1.0.0"
  }
}

    For production projects, prefer an immutable tag such as #1.0.0 over a branch name.

    Use A Local Checkout

    For SDK development or local validation, use Add package from disk... and select:

    Packages/com.vigram.sdk-gh1/package.json

    Service Lifetime Rule

    Vigram service members are factories. They do not return shared singleton instances for the active app session. Create a service once in the owning session or screen and pass the same instance around.

    This matters most for Bluetooth: create one IBluetoothService, scan with it, stop scanning with it, connect with it, pass it into Vigram.GPSService(...), and disconnect with it.

    First Run Flow

    This example shows the complete Unity lifecycle for the first viDoc connection. It uses MonoBehaviour only to demonstrate Start() and OnDestroy(); the SDK services themselves are ordinary C# objects.

    using System.Collections.Generic;
using UnityEngine;
using ViGram;

public class VidocFirstRunExample : MonoBehaviour
{
    private IBluetoothService bluetooth;
    private IPeripheralService peripheral;
    private bool isConnecting;

    private void Start()
    {
        Vigram.Initial("YOUR_SDK_TOKEN").Check(OnAuthenticated);
    }

    private void OnAuthenticated(AuthenticationResult result)
    {
        if (result is AuthenticationResult.Error error)
        {
            Debug.LogError($"viDoc SDK authorization failed: {error.Enum} {error.Message}");
            return;
        }

        bluetooth = Vigram.BluetoothService;
        bluetooth.State(state => Debug.Log($"Bluetooth state: {state}"));
        bluetooth.ObserveAvailableDevices(OnDevicesChanged);
        bluetooth.StartScan();
    }

    private void OnDevicesChanged(List<BPeripheral> devices)
    {
        if (isConnecting || devices.Count == 0)
        {
            return;
        }

        // Replace this with your device picker UI.
        Connect(devices[0]);
    }

    private void Connect(BPeripheral selectedPeripheral)
    {
        isConnecting = true;
        bluetooth.StopScan();
        bluetooth.Connect(selectedPeripheral);

        peripheral = Vigram.PeripheralService(selectedPeripheral);
        peripheral.State = state => Debug.Log($"Peripheral state: {state}");
        peripheral.ConfigurationState = state => Debug.Log($"Configuration: {state.GetType().Name}");
        peripheral.Nmea = OnNmeaMessage;
        peripheral.DeviceMessages = message => Debug.Log($"Device message: {message.GetType().Name}");
        peripheral.SatelliteMessages = message => Debug.Log($"Satellite message: {message.GetType().Name}");
        peripheral.Start();
    }

    private void OnNmeaMessage(NmeaMessage message)
    {
        switch (message)
        {
            case NmeaMessage.Gga gga:
                Debug.Log($"GGA coordinate: {gga.Coordinate?.Latitude}, {gga.Coordinate?.Longitude}");
                break;

            case NmeaMessage.Gst gst:
                Debug.Log($"GST accuracy: H={gst.Accuracy.Horizontal} V={gst.Accuracy.Vertical}");
                break;
        }
    }

    private void OnDestroy()
    {
        bluetooth?.StopScan();
        bluetooth?.Disconnect();
    }
}

    The SDK does not provide a device picker UI. ObserveAvailableDevices can fire multiple times while scanning, so production apps should show the latest list and let the user choose the intended BPeripheral.

    Initial Peripheral Configuration

    Pass PeripheralConfiguration to Vigram.PeripheralService(...) when the app needs a known startup profile for GNSS rate, elevation mask, constellation state, NAV-DOP/NAV-PVT messages, or dynamic model.

    PeripheralConfiguration configuration = new(
    rateOfChangeMessages: RateValue.Rate.Hertz7,
    elevationType: ElevationValue.Elevation.Angle10,
    glonassActivate: true,
    beidouActivate: true,
    galileoActivate: true,
    qzssActivate: true,
    sbassActivate: true,
    navDOPActivate: true,
    navPVTActivate: true,
    dynamicType: DynamicStateType.Pedestrian);

IPeripheralService peripheral = Vigram.PeripheralService(
    selectedPeripheral,
    configuration: configuration);

    Subscribe to ConfigurationState before calling Start() so the application can show progress or failure details.

    File Logging

    Create the logger before creating the logged peripheral service.

    using System.IO;
using UnityEngine;
using ViGram;

string logPath = Path.Combine(Application.persistentDataPath, "vidoc-session.csv");
ILogger logger = Vigram.LoggerService(logPath);
IPeripheralService peripheral = Vigram.PeripheralService(selectedPeripheral, logger: logger);

    Use logging during integration, field tests, and support sessions. For detailed logging guidance, see Log A Full Session.
